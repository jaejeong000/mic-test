<!DOCTYPE html>
<html>
<head>
  <meta charset="UTF-8">
  <title>Voice Recognition</title>
</head>
<body>
  <h2>음성 인식 중...</h2>
  <script>
    function startRecognition() {
      const recognition = new (window.SpeechRecognition || window.webkitSpeechRecognition)();
      recognition.lang = "en-US"; // 필요한 언어로 변경 가능
      recognition.onstart = () => {
        console.log("음성 인식 시작됨");
      };
      recognition.onresult = (event) => {
        const transcript = event.results[0][0].transcript;
        console.log("인식 결과:", transcript);
        // App Inventor로 결과 전달
        if (window.AppInventor) {
          window.AppInventor.setWebViewString(transcript);
        }
      };
      recognition.onerror = (event) => {
        console.error("인식 오류:", event.error);
        if (window.AppInventor) {
          window.AppInventor.setWebViewString("error: " + event.error);
        }
      };
      recognition.start();
    }
    window.onload = () => {
      startRecognition(); // 페이지 로딩되면 자동 실행
    };
  </script>
</body>
</html>
