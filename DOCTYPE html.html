<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <title>Mic Input</title>
</head>
<body>
  <h2>🎤 Speak Now...</h2>
  <p id="result">Waiting...</p>
  <script>
    const resultElement = document.getElementById('result');
    const recognition = new (window.SpeechRecognition || window.webkitSpeechRecognition)();
    recognition.lang = 'en-US'; // 영어 인식
    recognition.onresult = function(event) {
      const transcript = event.results[0][0].transcript;
      resultElement.textContent = transcript;
      window.AppInventor.setWebViewString(transcript); // App Inventor로 전송
    };
    recognition.onerror = function(event) {
      resultElement.textContent = 'Error: ' + event.error;
    };
    recognition.onend = function() {
      recognition.start(); // 끊기면 자동 재시작
    };
    recognition.start();
  </script>
</body>
</html>
